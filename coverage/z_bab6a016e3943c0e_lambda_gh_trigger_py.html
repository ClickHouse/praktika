47def verify_github_signature(event) -> None: 

48 if not WEBHOOK_SECRET: 

49 print("WARNING: GH_WEBHOOK_SECRET not set, skipping signature verification") 

50 return 

51 

52 headers = event.get("headers") or {} 

53 signature = headers.get("X-Hub-Signature-256") or headers.get( 

54 "x-hub-signature-256" 

55 ) 

56 if not signature: 

57 raise ValueError("Missing X-Hub-Signature-256 header") 

58 

59 raw_body = _get_raw_body(event) 

60 expected = ( 

61 "sha256=" 

62 + hmac.new( 

63 WEBHOOK_SECRET.encode("utf-8"), 

64 raw_body.encode("utf-8"), 

65 hashlib.sha256, 

66 ).hexdigest() 

67 ) 

68 

69 if not hmac.compare_digest(expected, signature): 

70 raise ValueError("Invalid GitHub webhook signature")

95def _build_push_workflow(payload, event_ts): 

96 """Build a CI workflow message from a push event. Returns None to skip 

97 (refs that aren't branch pushes, or branches not on the allow-list).""" 

98 ref = payload.get("ref", "") 

99 if not ref.startswith("refs/heads/"): 99 ↛ 100line 99 didn't jump to line 100 because the condition on line 99 was never true

100 return None 

101 branch = ref[len("refs/heads/") :] 

102 if branch not in ALLOWED_PUSH_BRANCHES: 

103 return None 

104 head_sha = payload.get("after") or payload.get("head_commit", {}).get("id", "") 

105 if not head_sha: 105 ↛ 106line 105 didn't jump to line 106 because the condition on line 105 was never true

106 return None 

107 return { 

108 "type": "push", 

109 "event_ts": event_ts, 

110 # head_ref carries the branch the push happened on. Match the 

111 # PR-event shape so every downstream consumer (orchestrator 

112 # workflow matcher, _dispatch's task builder, 

113 # _build_ci_environment) reads the same field name. 

114 "head_ref": branch, 

115 "head_sha": head_sha, 

116 "repo": payload.get("repository", {}).get("full_name", ""), 

117 "sender": payload.get("sender", {}).get("login", ""), 

118 }

121def _build_rerun_workflow(check_obj, payload, event_ts): 

122 """Build a rerun workflow message from a check_suite or check_run payload. 

123 

124 Returns (workflow_dict, pr_number) or (None, None) if the event has no 

125 associated PR (e.g. a push-based check suite with no PR). 

126 """ 

127 prs = check_obj.get("pull_requests", []) 

128 if not prs: 

129 return None, None 

130 pr = prs[0] 

131 pr_number = pr.get("number") 

132 head_sha = check_obj.get("head_sha", "") 

133 if not pr_number or not head_sha: 

134 return None, None 

135 workflow = { 

136 "type": "pull_request", 

137 "action": "rerequested", 

138 "event_ts": event_ts, 

139 "pr_number": pr_number, 

140 "head_sha": head_sha, 

141 "head_ref": pr.get("head", {}).get("ref", ""), 

142 "base_ref": pr.get("base", {}).get("ref", ""), 

143 "repo": payload.get("repository", {}).get("full_name", ""), 

144 "sender": payload.get("sender", {}).get("login", ""), 

145 "title": "", 

146 "draft": False, 

147 "labels": [], 

148 } 

149 return workflow, pr_number

178def _cancel_run(run_id): 

179 """Manual UI Cancel button: write per-run cancel-request to S3. 

180 

181 The orchestrator's ``sweep_cancel`` polls this key every wait() cycle, 

182 sets ``state.cancelled``, and the main loop drives the rest (cancel 

183 PENDING/RUNNING non-always_run jobs, write the runner kill flag). The 

184 object is small and only its presence matters; idempotent re-write on 

185 duplicate webhook deliveries is harmless. 

186 """ 

187 if not S3_BUCKET: 

188 print(" [warn] S3_BUCKET not set; cannot write cancel-request") 

189 return 

190 key = f"runs/{run_id}/cancel-request" 

191 try: 

192 _s3().put_object(Bucket=S3_BUCKET, Key=key, Body=b"requested") 

193 print(f"CANCEL request written: s3://{S3_BUCKET}/{key}") 

194 except Exception as e: 

195 print(f" [warn] could not write cancel-request: {e}")

198def _cancel_runs_before(pr_number, event_ts): 

199 """New push: write the scoped ``cancel-before`` marker with the new event ts. 

200 

201 Every still-running orchestrator for this PR in the same queue scope 

202 with ``event_ts < cancel-before`` self-cancels via ``sweep_cancel``. 

203 The freshly enqueued run carries the same ``event_ts`` so it stays 

204 alive (sweep_cancel uses strict less-than). 

205 """ 

206 if not S3_BUCKET: 

207 print(" [warn] S3_BUCKET not set; cannot write cancel-before") 

208 return 

209 key = _cancel_before_key(pr_number) 

210 try: 

211 _s3().put_object( 

212 Bucket=S3_BUCKET, 

213 Key=key, 

214 Body=json.dumps({"ts": event_ts}).encode(), 

215 ContentType="application/json", 

216 ) 

217 print(f"CANCEL-BEFORE written: s3://{S3_BUCKET}/{key} ts={event_ts:.0f}") 

218 except Exception as e: 

219 print(f" [warn] could not write cancel-before: {e}")

223 """Send a CI workflow trigger to SQS.""" 

224 _sqs().send_message(

226 MessageBody=json.dumps(workflow), 

227 MessageAttributes={ 

228 "delivery_id": {"DataType": "String", "StringValue": delivery_id}, 

229 }, 

230 ) 

231 label = workflow["type"] 

232 if workflow.get("action"): 

233 label += f".{workflow['action']}" 

234 target = ( 

235 f"PR#{workflow['pr_number']}" 

236 if workflow.get("pr_number") 

237 else f"branch={workflow.get('head_ref', '?')}" 

238 ) 

239 print(f"ENQUEUED: {label} {target} delivery={delivery_id}")

242def lambda_handler(event, context): 

243 try: 

244 verify_github_signature(event) 

245 except Exception as e: 

246 print(f"Signature verification failed: {e}") 

247 return {"statusCode": 401, "body": "unauthorized"} 

248 

249 # One ts per webhook delivery: stamped on the workflow event AND used 

250 # as the scoped cancel-before marker for older runs of the same PR. 

251 # Sequential operations in this invocation share the same ts so the 

252 # freshly enqueued run (event_ts == cancel-before-ts) does not 

253 # self-cancel. 

254 event_ts = time.time() 

255 

256 headers = event.get("headers") or {} 

257 gh_event = headers.get("X-GitHub-Event") or headers.get("x-github-event", "unknown") 

258 delivery_id = headers.get("X-GitHub-Delivery") or headers.get( 

259 "x-github-delivery", "" 

260 ) 

261 

262 raw_body = _get_raw_body(event) 

263 try: 

264 payload = json.loads(raw_body) 

265 except (json.JSONDecodeError, TypeError): 

266 payload = {} 

267 

268 action = payload.get("action", "") 

269 sender = payload.get("sender", {}).get("login", "") 

270 

271 print(f"EVENT: {gh_event}.{action} SENDER: {sender} DELIVERY: {delivery_id}") 

272 

273 if gh_event == "check_run": 

274 if action == "requested_action": 

275 identifier = payload.get("requested_action", {}).get("identifier", "") 

276 if identifier == "cancel": 

277 check_run = payload.get("check_run", {}) 

278 head_sha = check_run.get("head_sha", "") 

279 prs = check_run.get("pull_requests", []) 

280 pr_number = prs[0].get("number") if prs else None 

281 # check_run.id is the orchestrator run_id (per-run S3 prefix). 

282 run_id = str(check_run.get("id", "")) or None 

283 if run_id: 

284 _cancel_run(run_id) 

285 print(f"MANUAL CANCEL: PR#{pr_number} run_id={run_id} sha={head_sha[:12]}") 

286 else: 

287 print(f"SKIP: cancel action missing check_run id") 

288 elif action == "rerequested": 

289 workflow, pr_number = _build_rerun_workflow( 

290 payload.get("check_run", {}), payload, event_ts 

291 ) 

292 if workflow and (not ALLOWED_SENDERS or sender in ALLOWED_SENDERS): 

293 # No cancel: re-run always targets the same SHA, so a 

294 # target_sha cancel would be consumed by the new orchestrator 

295 # and cancel itself. The previous run for that SHA has already 

296 # finished — there is nothing to cancel. 

297 _enqueue(workflow, delivery_id) 

298 print(f"RERUN (check_run): PR#{pr_number} sha={workflow['head_sha'][:12]}") 

299 else: 

300 print("SKIP: check_run.rerequested — no associated PR or sender not allowed") 

301 else: 

302 print(f"SKIP: check_run.{action} not handled") 

303 return {"statusCode": 200, "body": "ok"} 

304 

305 if gh_event == "check_suite": 

306 if action == "rerequested": 

307 workflow, pr_number = _build_rerun_workflow( 

308 payload.get("check_suite", {}), payload, event_ts 

309 ) 

310 if workflow and (not ALLOWED_SENDERS or sender in ALLOWED_SENDERS): 

311 # No cancel: re-run targets the same SHA as before but 

312 # spawns a new run_id (new check run), so the new run has 

313 # its own S3 prefix and can't conflict with any prior run. 

314 _enqueue(workflow, delivery_id) 

315 print(f"RERUN (check_suite): PR#{pr_number} sha={workflow['head_sha'][:12]}") 

316 else: 

317 print("SKIP: check_suite.rerequested — no associated PR or sender not allowed") 

318 else: 

319 print(f"SKIP: check_suite.{action} not handled") 

320 return {"statusCode": 200, "body": "ok"} 

321 

322 if gh_event == "push": 

323 if ALLOWED_SENDERS and sender not in ALLOWED_SENDERS: 

324 print(f"SKIP: push sender {sender} not in allowed list") 

325 return {"statusCode": 200, "body": "ok"} 

326 workflow = _build_push_workflow(payload, event_ts) 

327 if workflow: 

328 _enqueue(workflow, delivery_id) 

329 print( 

330 f"PUSH: branch={workflow['head_ref']} sha={workflow['head_sha'][:12]}" 

331 ) 

332 else: 

333 ref = payload.get("ref", "") 

334 print(f"SKIP: push ref {ref} not on the allow-list") 

335 return {"statusCode": 200, "body": "ok"} 

336 

337 if gh_event != "pull_request": 

338 print(f"SKIP: not a pull_request event") 

339 return {"statusCode": 200, "body": "ok"} 

340 

341 if ALLOWED_SENDERS and sender not in ALLOWED_SENDERS: 

342 print(f"SKIP: sender {sender} not in allowed list") 

343 return {"statusCode": 200, "body": "ok"} 

344 

345 workflow = _build_workflow(action, payload, event_ts) 

346 if workflow: 

347 # On a new push (synchronize) write the scoped cancel-before marker 

348 # with this event_ts BEFORE enqueuing the new run, so older runs in 

349 # the same orchestrator scope see the flag on their next 

350 # sweep_cancel and self-cancel. The freshly enqueued run carries 

351 # the same event_ts and stays alive. 

352 if action == "synchronize": 

353 _cancel_runs_before(workflow["pr_number"], event_ts) 

354 _enqueue(workflow, delivery_id) 

355 else: 

356 print(f"SKIP: action {action} does not trigger a workflow") 

357 

358 return { 

359 "statusCode": 200, 

360 "headers": {"Content-Type": "application/json"}, 

361 "body": json.dumps({ 

362 "ok": True, 

363 "event": gh_event, 

364 "action": action, 

365 "enqueued": workflow is not None, 

366 }), 

367 }
