82def build_job_dag(workflow): 

83 """Build the dependency DAG for a workflow's jobs. 

84 

85 Returns (levels, job_deps) where: 

86 - levels: list of lists, each inner list is a set of jobs that can run in parallel 

87 - job_deps: dict mapping job name -> set of job names it depends on 

88 """ 

89 jobs_by_name = {job.name: job for job in workflow.jobs} 

90 artifact_provider = _get_artifact_to_providing_job_map(workflow) 

91 

92 # Build adjacency: job_name -> set of predecessor job names 

93 job_deps = defaultdict(set) 

94 for job in workflow.jobs: 

95 # Hard deps: requires -> find the providing job 

96 for req in job.requires: 

97 if req in artifact_provider: 

98 job_deps[job.name].add(artifact_provider[req]) 

99 elif req in jobs_by_name: 

100 # requires can also reference job names directly (for report artifacts) 

101 job_deps[job.name].add(req) 

102 # Ordering deps 

103 for dep_name in job.run_after: 

104 if dep_name in jobs_by_name: 

105 job_deps[job.name].add(dep_name) 

106 # Ensure every job appears in the dict 

107 job_deps.setdefault(job.name, set()) 

108 

109 # Topological sort by levels (Kahn's algorithm) 

110 in_degree = {name: len(deps) for name, deps in job_deps.items()} 

111 # Reverse: who depends on me 

112 dependents = defaultdict(set) 

113 for name, deps in job_deps.items(): 

114 for dep in deps: 

115 dependents[dep].add(name) 

116 

117 levels = [] 

118 ready = deque(sorted(n for n, d in in_degree.items() if d == 0)) 

119 visited = set() 

120 

121 while ready: 

122 level = list(ready) 

123 levels.append(level) 

124 ready.clear() 

125 for name in level: 

126 visited.add(name) 

127 for succ in sorted(dependents[name]): 

128 in_degree[succ] -= 1 

129 if in_degree[succ] == 0: 

130 ready.append(succ) 

131 

132 # Check for cycles 

133 if len(visited) != len(jobs_by_name): 

134 missing = set(jobs_by_name) - visited 

135 print(f"WARNING: dependency cycle detected, unreachable jobs: {missing}") 

136 

137 return levels, dict(job_deps)
